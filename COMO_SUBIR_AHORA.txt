🔧 SI HAY PROBLEMAS:

Problema: "No se puede hacer push"
Solución:
  git pull origin main --rebase
  git push origin main

Problema: "Conflictos al hacer pull"
Solución:
  # Edita archivos con conflictos
  git add .
  git commit -m "resolve conflicts"
  git push origin main

Problema: ".env se está subiendo"
Solución:
  git rm --cached .env
  echo ".env" >> .gitignore
  git add .gitignore
  git commit -m "fix: protect .env file"
  git push origin main

Problema: "No hay remote configurado"
Solución:
  git remote add origin https://github.com/TU_USUARIO/TU_REPO.git
  git push -u origin main
